Q.
다음 프로그램의 실행 결과는 무엇인가?
using System;
enum Color {Red, Green, Blue, Max = Blue};
class ExerciseCh2_10_2 {
    public static void Main()
    {
        Color c = Color.Red;
        int i = (int)++c;
        Console.WriteLine(" Color = " + (Color)i);
        c = Color.Blue;
        Console.WriteLine("Cardinality of " + c + " = " + (int)c);
        c = Color.Max;
        Console.WriteLine("Cardinality of " + c + " = " + (int)c);
    }
}

A.
 Color = Green
Cardinality of Blue = 2
Cardinality of Blue = 2
